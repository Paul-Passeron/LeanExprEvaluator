\section{Grammar}
We'll define the set of arithmetic expressions as the expressions that can be built from the following grammar:
\begin{align*}
E_A ::= n (\in \mathbb{Z}) \mid E_A + E_A \mid E_A \times E_A \mid E_A - E_A
\end{align*}

\section{Small Steps Semantic}
\bigskip
\bigskip
\begin{prooftree}
\AxiomC{}
\LeftLabel{($A_1$)}
\RightLabel{
\begin{pmatrix}
    \text{if } n_1, n_2 \in \mathbb{Z},\\
    \text{op } \in \left\{ +, -, \times \right\}\\
    \text{and } n \in \mathbb{Z}, \hspace{.5em} n = n_1 \text{ op } n_2
\end{pmatrix}}
\UnaryInfC{$\langle n_1 \text{ op } n_2 \rangle \hookrightarrow \langle n \rangle$}
\end{prooftree}

\bigskip
\bigskip

\begin{prooftree}
\AxiomC{$\langle e_1 \rangle \hookrightarrow \langle e'_1 \rangle$}
\LeftLabel{($A_2$)}
\RightLabel{
\begin{pmatrix}
\text{if op } \in \left\{ +, -, \times \right\}
\end{pmatrix}
}
\UnaryInfC{$\langle e_1 \text{ op } e_2 \rangle \hookrightarrow \langle e'_1 \text{ op } e_2 \rangle$}
\end{prooftree}
